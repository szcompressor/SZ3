44 size_t compress(const Config &conf, T *data, uchar *cmpData, size_t cmpCap) override {
45 std::vector<int> quant_inds = decomposition.compress(conf, data);
46
47 if (decomposition.get_out_range().first != 0) {
48 throw std::runtime_error("The output range of the decomposition must start from 0 for this compressor");
49 }
50 encoder.preprocess_encode(quant_inds, decomposition.get_out_range().second);
51 size_t bufferSize = std::max<size_t>(
52 1000, 2 * (decomposition.size_est() + encoder.size_est() + sizeof(T) * quant_inds.size()));
53
54 auto buffer = static_cast<uchar *>(malloc(bufferSize));
55 uchar *buffer_pos = buffer;
56
57 decomposition.save(buffer_pos);
58 encoder.save(buffer_pos);
59
60 //store the size of quant_inds is necessary as it is not always equal to conf.num
61 write<size_t>(quant_inds.size(), buffer_pos);
62 encoder.encode(quant_inds, buffer_pos);
63 encoder.postprocess_encode();
64
65 auto cmpSize = lossless.compress(buffer, buffer_pos - buffer, cmpData, cmpCap);
66 free(buffer);
67
68 return cmpSize;
69 }

71 T *decompress(const Config &conf, uchar const *cmpData, size_t cmpSize, T *decData) override {
72 uchar *buffer = nullptr;
73 size_t bufferSize = 0;
74 lossless.decompress(cmpData, cmpSize, buffer, bufferSize);
75
76 uchar const *bufferPos = buffer;
77
78 decomposition.load(bufferPos, bufferSize);
79 encoder.load(bufferPos, bufferSize);
80
81 size_t quant_inds_size = 0;
82 read(quant_inds_size, bufferPos);
83 auto quant_inds = encoder.decode(bufferPos, quant_inds_size);
84 encoder.postprocess_decode();
85
86 free(buffer);
87
88 decomposition.decompress(conf, quant_inds, decData);
89 return decData;
90 }
