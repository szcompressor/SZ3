98 int cal_site(T *data, std::vector<size_t> dims) {
99 size_t numDims = dims.size();
100 if (numDims < 2) {
101 return 0;
102 }
103 std::vector<int> sites;
104 for (size_t j = 0; j < std::min<size_t>(dims[numDims - 1], 5); j++) {
105 // size_t lprev = 0, lavg = 0, lcnt = 0;
106 size_t lprev = 0;
107 for (size_t i = 1; i < std::min<size_t>(dims[numDims - 2], 100); i++) {
108 auto c = data[i * dims[numDims - 1] + j], p = data[(i - 1) * dims[numDims - 1] + j];
109 if (fabs(c - p) / c > 0.5) {
110 sites.push_back(i - lprev);
111 // printf("%d %d\n", i, i - lprev);
112 lprev = i;
113 }
114 }
115 }
116 unordered_map<int, size_t> frequency;
117 for (size_t i = 0; i < sites.size(); i++) {
118 frequency[sites[i]]++;
119 }
120 size_t maxCount = 0;
121 int res = 0;
122 for (const auto &kv : frequency) {
123 auto k = kv.first;
124 auto f = kv.second;
125 // printf("k %d f %d\n", k ,f);
126 if (maxCount < f) {
127 res = k;
128 maxCount = f;
129 }
130 }
131 return (res <= 2 || res > 10) ? 0 : res;
132 }

134 /* Start from the last frame and look for frames filled with the same value.
135 * Those frames do not need to be compressed - they can all just be filled. */
136 std::tuple<size_t, T> findFillValueAndFirstFilledFrame(T *data, std::vector<size_t> dims) {
137 size_t numDims = dims.size();
138 if (numDims < 3) {
139 return std::make_tuple(dims[0], 0);
140 }
141 size_t frameStride = dims[1] * dims[2];
142 size_t firstFillFrame = dims[0];
143
144 if (firstFillFrame == 0) {
145 return std::make_tuple(firstFillFrame, 0);
146 }
147 /* Assume that the first value of the last frame is the potential fill value */
148 T fillFrameValue = data[(dims[0] - 1) * frameStride];
149 /* To simplify compression/decompression below, assume that the first frame needs
150 * to be compressed. */
151 for (size_t i = dims[0] - 1; i > 0; i--) {
152 size_t idx = i * frameStride;
153 bool allFrameValuesAreFillValue = true;
154
155 for (size_t j = 0; j < dims[1] * dims[2]; j++) {
156 size_t idy = idx + j;
157 if (data[idy] != fillFrameValue) {
158 allFrameValuesAreFillValue = false;
159 break;
160 }
161 }
162 if (allFrameValuesAreFillValue) {
163 firstFillFrame = i;
164 } else {
165 break;
166 }
167 }
168 return std::make_tuple(firstFillFrame, fillFrameValue);
169 }

171 std::vector<int> compress_2d(T *data) {
172 std::vector<int> quant_bins(conf.num);
173 auto dims = conf.dims;
174 site = cal_site(data, conf.dims);
175 // printf("# of site in the MD simulation guessed by SZ3 = %d\n", site);
176
177 firstFillFrame_ = dims[0];
178 fillValue_ = 0;
179
180 // i==0 & j==0
181 for (size_t k = 0; k < dims[1]; k++) { // xyz
182 size_t idx = k;
183 quant_bins[idx] = quantizer.quantize_and_overwrite(data[idx], 0);
184 }
185
186 // i==0
187 for (size_t j = 1; j < dims[0]; j++) { // atoms
188 /* Assume that we are compressing water coordinates. Water models are constructed the following way:
189 * 3-site: O-H-H
190 * 4-site: O-H-H-VS
191 * 5-site: O-H-H-VS-VS
192 * The oxygen atom is always most centrally located (on average closest to the others).
193 * Use the oxygen atom as reference.
194 */
195 size_t site_reference_offset = site != 0 ? std::max<size_t>(1, j % site) : 1;
196 for (size_t k = 0; k < dims[1]; k++) { // xyz
197 size_t idx = j * dims[1] + k;
198 size_t idx1 = (j - site_reference_offset) * dims[1] + k;
199 quant_bins[idx] = quantizer.quantize_and_overwrite(data[idx], data[idx1]);
200 }
201 }
202
203 return quant_bins;
204 }

235 std::vector<int> compress_3d(T *data) {
236 std::vector<int> quant_bins(conf.num);
237 auto dims = conf.dims;
238 std::vector<size_t> stride({dims[1] * dims[2], dims[2], 1});
239 site = cal_site(data + stride[0], conf.dims);
240 // printf("# of site in the MD simulation guessed by SZ3 = %d\n", site);
241
242 /* Find out if the last frames are all filled with the same value. */
243 std::tuple<size_t, T> fillValueSettings = findFillValueAndFirstFilledFrame(data, dims);
244 firstFillFrame_ = std::get<0>(fillValueSettings);
245 fillValue_ = std::get<1>(fillValueSettings);
246 size_t lastFrame = std::min(dims[0], firstFillFrame_);
247
248 // TODO determine the # of system
249 // i==0 & j==0
250 for (size_t k = 0; k < dims[2]; k++) { // xyz
251 size_t idx = k;
252 quant_bins[idx] = quantizer.quantize_and_overwrite(data[idx], 0);
253 }
254
255 // i==0
256 for (size_t j = 1; j < dims[1]; j++) { // atoms
257 /* Assume that we are compressing water coordinates. Water models are constructed the following way:
258 * 3-site: O-H-H
259 * 4-site: O-H-H-VS
260 * 5-site: O-H-H-VS-VS
261 * The oxygen atom is always most centrally located (on average closest to the others).
262 * Use the oxygen atom as reference.
263 */
264 size_t site_reference_offset = site != 0 ? std::max<size_t>(1, j % site) : 1;
265 for (size_t k = 0; k < dims[2]; k++) { // xyz
266 size_t idx = j * stride[1] + k;
267 size_t idx1 = (j - site_reference_offset) * stride[1] + k;
268 quant_bins[idx] = quantizer.quantize_and_overwrite(data[idx], data[idx1]);
269 }
270 }
271
272 for (size_t i = 1; i < lastFrame; i++) { // time
273 for (size_t j = 0; j < dims[1]; j++) { // atoms
274 size_t site_reference_offset = site != 0 ? j % site : 1;
275 for (size_t k = 0; k < dims[2]; k++) { // xyz
276 size_t idx = i * stride[0] + j * stride[1] + k;
277 size_t idx1 = (i - 1) * stride[0] + j * stride[1] + k;

281 size_t idx2 = i * stride[0] + (j - site_reference_offset) * stride[1] + k;
282 size_t idx3 = (i - 1) * stride[0] + (j - site_reference_offset) * stride[1] + k;
283 quant_bins[idx] =

285 }

327 size_t idx = i * stride[0] + j * stride[1] + k;

329 if (j == 0 || (site != 0 && j % site == 0)) { // time -1
330 dec_data[idx] = quantizer.recover(dec_data[idx1], quant_inds[idx]);

337 }
338 }
339 }
