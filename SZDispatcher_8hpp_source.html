13template <class T, uint N>
14size_t SZ_compress_dispatcher(Config &conf, const T *data, uchar *cmpData, size_t cmpCap) {
15 if (conf.cmprAlgo == ALGO_SVD && std::is_integral<T>::value) {
16 throw std::invalid_argument("SVD algorithm only supports floating-point data types.");
17 }
18 assert(N == conf.N);
19 calAbsErrorBound(conf, data);
20 size_t cmpSize = 0;
21
22 // if absErrorBound is 0, use lossless only mode
23 if (conf.absErrorBound == 0) {
24 conf.cmprAlgo = ALGO_LOSSLESS;
25 }
26
27 // do lossy compression
28 bool isCmpCapSufficient = true;
29 if (conf.cmprAlgo != ALGO_LOSSLESS) {
30 try {
31 std::vector<T> dataCopy(data, data + conf.num);
32 if (conf.cmprAlgo == ALGO_LORENZO_REG) {
33 cmpSize = SZ_compress_LorenzoReg<T, N>(conf, dataCopy.data(), cmpData, cmpCap);
34 } else if (conf.cmprAlgo == ALGO_INTERP) {
35 cmpSize = SZ_compress_Interp<T, N>(conf, dataCopy.data(), cmpData, cmpCap);
36 } else if (conf.cmprAlgo == ALGO_INTERP_LORENZO) {
37 cmpSize = SZ_compress_Interp_lorenzo<T, N>(conf, dataCopy.data(), cmpData, cmpCap);
38 } else if (conf.cmprAlgo == ALGO_NOPRED) {
39 cmpSize = SZ_compress_nopred<T, N>(conf, dataCopy.data(), cmpData, cmpCap);
40 } else if (conf.cmprAlgo == ALGO_BIOMD) {
41 return SZ_compress_bioMD<T, N>(conf, dataCopy.data(), cmpData, cmpCap);
42 } else if (conf.cmprAlgo == ALGO_BIOMDXTC) {
43 return SZ_compress_bioMDXtcBased<T, N>(conf, dataCopy.data(), cmpData, cmpCap);
44 } else if (conf.cmprAlgo == ALGO_SVD) {
45 cmpSize = SZ_compress_SVD<T, N>(conf, dataCopy.data(), cmpData, cmpCap);
46 } else {
47 throw std::invalid_argument("Unknown compression algorithm");
48 }
49
50 } catch (std::length_error &e) {
51 if (std::string(e.what()) == SZ3_ERROR_COMP_BUFFER_NOT_LARGE_ENOUGH) {
52 isCmpCapSufficient = false;
53 // printf("SZ is downgraded to lossless mode because the buffer for compressed data is not large enough.\n");
54 } else {
55 throw;
56 }
57 }
58 }
59
60 // do lossless only compression if 1) cmpr algorithm is lossless or 2) compressed buffer not large enough for lossy
61 if (conf.cmprAlgo == ALGO_LOSSLESS || !isCmpCapSufficient) {
62 conf.cmprAlgo = ALGO_LOSSLESS;
63 auto zstd = Lossless_zstd();
64 return zstd.compress(reinterpret_cast<const uchar *>(data), conf.num * sizeof(T), cmpData, cmpCap);
65 }
66
67 // if lossy compression ratio < 3, test if lossless only mode has a better ratio than lossy
68 if (conf.num * sizeof(T) / 1.0 / cmpSize < 3) {
69 auto zstd = Lossless_zstd();
70 auto zstdCmpCap = ZSTD_compressBound(conf.num * sizeof(T)) + sizeof(size_t);
71 auto zstdCmpData = static_cast<uchar *>(malloc(zstdCmpCap));
72 size_t zstdCmpSize =
73 zstd.compress(reinterpret_cast<const uchar *>(data), conf.num * sizeof(T), zstdCmpData, zstdCmpCap);
74 if (zstdCmpSize < cmpSize && zstdCmpSize <= cmpCap) {
75 conf.cmprAlgo = ALGO_LOSSLESS;
76 memcpy(cmpData, zstdCmpData, zstdCmpSize);
77 cmpSize = zstdCmpSize;
78 }
79 free(zstdCmpData);
80 }
81 return cmpSize;
82}

84template <class T, uint N>
85void SZ_decompress_dispatcher(Config &conf, const uchar *cmpData, size_t cmpSize, T *decData) {
86 if (conf.cmprAlgo == ALGO_LOSSLESS) {
87 auto zstd = Lossless_zstd();
88 size_t decDataSize = 0;
89 auto decDataPos = reinterpret_cast<uchar *>(decData);
90 zstd.decompress(cmpData, cmpSize, decDataPos, decDataSize);
91 if (decDataSize != conf.num * sizeof(T)) {
92 throw std::runtime_error("Decompressed data size does not match the original data size");
93 }
94 } else if (conf.cmprAlgo == ALGO_LORENZO_REG) {
95 SZ_decompress_LorenzoReg<T, N>(conf, cmpData, cmpSize, decData);
96 } else if (conf.cmprAlgo == ALGO_INTERP) {
97 SZ_decompress_Interp<T, N>(conf, cmpData, cmpSize, decData);
98 } else if (conf.cmprAlgo == ALGO_NOPRED) {
99 SZ_decompress_nopred<T, N>(conf, cmpData, cmpSize, decData);
100 } else if (conf.cmprAlgo == ALGO_BIOMD) {
101 SZ_decompress_bioMD<T, N>(conf, cmpData, cmpSize, decData);
102 } else if (conf.cmprAlgo == ALGO_BIOMDXTC) {
103 SZ_decompress_bioMDXtcBased<T, N>(conf, cmpData, cmpSize, decData);
104 } else if (conf.cmprAlgo == ALGO_SVD) {
105 SZ_decompress_SVD<T, N>(conf, cmpData, cmpSize, decData);
106 } else {
107 throw std::invalid_argument("Unknown compression algorithm");
108 }
109}
