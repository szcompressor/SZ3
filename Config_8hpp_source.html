163 template <class Iter>
164 size_t setDims(Iter begin, Iter end) {
165 auto dims_ = std::vector<size_t>(begin, end);
166 dims.clear();
167 for (auto dim : dims_) {
168 if (dim > 1) {
169 dims.push_back(dim);
170 }
171 }
172 if (dims.empty()) {
173 dims = {1};
174 }
175 N = dims.size();
176 num = std::accumulate(dims.begin(), dims.end(), static_cast<size_t>(1), std::multiplies<size_t>());
177 predDim = N;
178 blockSize = (N == 1 ? 128 : (N == 2 ? 16 : 6));
179 return num;
180 }

203 void load_ini(const std::string& ini_content) {
204 std::istringstream ss(ini_content);
205 std::string line, section;
206
207 auto trim = [](std::string& s) {
208 s.erase(0, s.find_first_not_of(" \t\r\n"));
209 s.erase(s.find_last_not_of(" \t\r\n") + 1);
210 };
211
212 auto eq = [&](const std::string& a, const std::string& b) { return to_lower(a) == to_lower(b); };
213
214 auto parse_bool = [&](const std::string& s) {
215 auto ls = to_lower(s);
216 return ls == "true" || ls == "1" || ls == "yes" || ls == "on";
217 };
218
219 while (std::getline(ss, line)) {
220 trim(line);
221 if (line.empty() || line[0] == '#') continue;
222 if (line.front() == '[') {
223 section = line.substr(1, line.find(']') - 1);
224 continue;
225 }
226
227 auto sep = line.find('=');
228 if (sep == std::string::npos) continue;
229
230 std::string key = line.substr(0, sep);
231 std::string value = line.substr(sep + 1);
232 trim(key);
233 trim(value);
234
235 if (eq(section, "GlobalSettings")) {
236 if (eq(key, "CmprAlgo"))
237 match_enum(value, ALGO_MAP, cmprAlgo);
238 else if (eq(key, "ErrorBoundMode"))
239 match_enum(value, EB_MAP, errorBoundMode);
240 else if (eq(key, "AbsErrorBound"))
241 absErrorBound = std::stod(value);
242 else if (eq(key, "RelErrorBound"))
243 relErrorBound = std::stod(value);
244 else if (eq(key, "PSNRErrorBound"))
245 psnrErrorBound = std::stod(value);
246 else if (eq(key, "L2NormErrorBound"))
247 l2normErrorBound = std::stod(value);
248 else if (eq(key, "OpenMP"))
249 openmp = parse_bool(value);
250 } else if (eq(section, "AlgoSettings")) {
251 if (eq(key, "Lorenzo"))
252 lorenzo = parse_bool(value);
253 else if (eq(key, "Lorenzo2ndOrder"))
254 lorenzo2 = parse_bool(value);
255 else if (eq(key, "Regression"))
256 regression = parse_bool(value);
257 else if (eq(key, "Regression2ndOrder"))
258 regression2 = parse_bool(value);
259 else if (eq(key, "InterpolationAlgo"))
260 match_enum(value, INTERP_ALGO_MAP, interpAlgo);
261 else if (eq(key, "InterpolationDirection"))
262 interpDirection = std::stoi(value);
263 else if (eq(key, "BlockSize"))
264 blockSize = std::stoi(value);
265 else if (eq(key, "QuantizationBinTotal"))
266 quantbinCnt = std::stoi(value);
267 else if (eq(key, "InterpolationAnchorStride"))
268 interpAnchorStride = std::stoi(value);
269 else if (eq(key, "InterpolationAlpha"))
270 interpAlpha = std::stod(value);
271 else if (eq(key, "InterpolationBeta"))
272 interpBeta = std::stod(value);
273 else if (eq(key, "SVDTargetRank"))
274 svd_target_rank = std::stoi(value);
275 else if (eq(key, "SVDOversamplingParam"))
276 svd_oversampling_param = std::stoi(value);
277 else if (eq(key, "SVDEnergyThreshold"))
278 svd_energy_threshold = std::stod(value);
279 else if (eq(key, "SVDQuantEBScale"))
280 svd_quant_eb_scale = std::stod(value);
281 }
282 }

327 size_t save(unsigned char*& c) const {
328 auto c0 = c;
329 c += sizeof(uchar); // reserve space for conf size
330
331 write(N, c);
332 // write(dims.data(), dims.size(), c);
333 auto bitWidth = vector_bit_width(dims);
334 write(bitWidth, c);
335 vector2bytes(dims, bitWidth, c);
336
337 write(num, c);
338 write(cmprAlgo, c);
339
340 write(errorBoundMode, c);
341 if (errorBoundMode == EB_ABS) {
342 write(absErrorBound, c);
343 } else if (errorBoundMode == EB_REL) {
344 write(relErrorBound, c);
345 } else if (errorBoundMode == EB_PSNR) {
346 write(psnrErrorBound, c);
347 } else if (errorBoundMode == EB_L2NORM) {
348 write(l2normErrorBound, c);
349 } else if (errorBoundMode == EB_ABS_OR_REL) {
350 write(absErrorBound, c);
351 write(relErrorBound, c);
352 } else if (errorBoundMode == EB_ABS_AND_REL) {
353 write(absErrorBound, c);
354 write(relErrorBound, c);
355 }
356
357 uint8_t boolvals = (lorenzo & 1) << 7 | (lorenzo2 & 1) << 6 | (regression & 1) << 5 | (regression2 & 1) << 4 |
358 (openmp & 1) << 3;
359 write(boolvals, c);
360
361 write(dataType, c);
362 write(quantbinCnt, c);
363 write(blockSize, c);
364 write(predDim, c);
365
366 // SVD specific parameters
367 write(svd_target_rank, c);
368 write(svd_oversampling_param, c);
369 write(svd_energy_threshold, c);
370 write(svd_quant_eb_scale, c);
371
372 auto confSize = static_cast<uchar>(c - c0);
373 write(confSize, c0); // write conf size at reserved space
374 return confSize;
375 }

382 void load(const unsigned char*& c) {
383 uchar confSize = 0;
384 read(confSize, c);
385 auto c1 = c + confSize;
386
387 read(N, c);
388 uint8_t bitWidth;
389 read(bitWidth, c);
390 dims = bytes2vector<size_t>(c, bitWidth, N);
391 // dims.resize(N);
392 // read(dims.data(), N, c);
393 read(num, c);
394 read(cmprAlgo, c);
395
396 read(errorBoundMode, c);
397 if (errorBoundMode == EB_ABS) {
398 read(absErrorBound, c);
399 } else if (errorBoundMode == EB_REL) {
400 read(relErrorBound, c);
401 } else if (errorBoundMode == EB_PSNR) {
402 read(psnrErrorBound, c);
403 } else if (errorBoundMode == EB_L2NORM) {
404 read(l2normErrorBound, c);
405 } else if (errorBoundMode == EB_ABS_OR_REL) {
406 read(absErrorBound, c);
407 read(relErrorBound, c);
408 } else if (errorBoundMode == EB_ABS_AND_REL) {
409 read(absErrorBound, c);
410 read(relErrorBound, c);
411 }
412
413 if (c < c1) {
414 uint8_t boolvals;
415 read(boolvals, c);
416 lorenzo = (boolvals >> 7) & 1;
417 lorenzo2 = (boolvals >> 6) & 1;
418 regression = (boolvals >> 5) & 1;
419 regression2 = (boolvals >> 4) & 1;
420 openmp = (boolvals >> 3) & 1;
421 }
422 if (c < c1) {
423 read(dataType, c);
424 }
425 if (c < c1) {
426 read(quantbinCnt, c);
427 }
428 if (c < c1) {
429 read(blockSize, c);
430 }
431 if (c < c1) {
432 read(predDim, c);
433 }
434 // SVD specific parameters
435 if (c < c1) {
436 read(svd_target_rank, c);
437 }
438 if (c < c1) {
