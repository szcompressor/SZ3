182 void init() {
183 quant_index = 0;
184 assert(blocksize % 2 == 0 && "Interpolation block size should be even numbers");
185 assert((anchor_stride & anchor_stride - 1) == 0 && "Anchor stride should be 0 or 2's exponentials");
186 num_elements = 1;
187 interp_level = -1;
188 bool use_anchor = false;
189 for (uint i = 0; i < N; i++) {
190 if (interp_level < ceil(log2(original_dimensions[i]))) {
191 interp_level = static_cast<int>(ceil(log2(original_dimensions[i])));
192 }
193 if (original_dimensions[i] > anchor_stride)
194 use_anchor = true;
195 num_elements *= original_dimensions[i];
196 }
197 if (!use_anchor)
198 anchor_stride = 0;
199 if (anchor_stride > 0) {
200 int max_interpolation_level = static_cast<int>(log2(anchor_stride)) + 1;
201 if (max_interpolation_level <= interp_level) {
202 interp_level = max_interpolation_level;
203 }
204 }
205
206 original_dim_offsets[N - 1] = 1;
207 for (int i = N - 2; i >= 0; i--) {
208 original_dim_offsets[i] = original_dim_offsets[i + 1] * original_dimensions[i + 1];
209 }

213 for (uint i = 0; i < N; i++) {

215 }
216 do {
217 dim_sequences.push_back(sequence);
218 } while (std::next_permutation(sequence.begin(), sequence.end()));

253 template <class QuantizeFunc>
254 double interpolation_1d(T *data, size_t begin, size_t end, size_t stride, const std::string &interp_func,
255 QuantizeFunc &&quantize_func) {
256 size_t n = (end - begin) / stride + 1;
257 if (n <= 1) {
258 return 0;
259 }
260 double predict_error = 0;
261
262 size_t stride3x = 3 * stride;
263 size_t stride5x = 5 * stride;
264 if (interp_func == "linear" || n < 5) {
265 // if (pb == PB_predict_overwrite) {
266 for (size_t i = 1; i + 1 < n; i += 2) {
267 T *d = data + begin + i * stride;
268 quantize_func(d - data, *d, interp_linear(*(d - stride), *(d + stride)));
269 }
270 if (n % 2 == 0) {
271 T *d = data + begin + (n - 1) * stride;
272 if (n < 4) {
273 quantize_func(d - data, *d, *(d - stride));
274 } else {
275 quantize_func(d - data, *d, interp_linear1(*(d - stride3x), *(d - stride)));
276 }
277 }
278 // }
279 } else {
280 T *d;
281 size_t i;
282 for (i = 3; i + 3 < n; i += 2) {
283 d = data + begin + i * stride;
284 quantize_func(d - data, *d,
285 interp_cubic(*(d - stride3x), *(d - stride), *(d + stride), *(d + stride3x)));
286 }
287 d = data + begin + stride;
288 quantize_func(d - data, *d, interp_quad_1(*(d - stride), *(d + stride), *(d + stride3x)));
289
290 d = data + begin + i * stride;
291 quantize_func(d - data, *d, interp_quad_2(*(d - stride3x), *(d - stride), *(d + stride)));
292 if (n % 2 == 0) {
293 d = data + begin + (n - 1) * stride;
294 quantize_func(d - data, *d, interp_quad_3(*(d - stride5x), *(d - stride3x), *(d - stride)));
295 }
296 }
297
298 return predict_error;
299 }

315 template <class QuantizeFunc>
316 double interpolation_1d_fastest_dim_first(T *data, const std::array<size_t, N> &begin_idx,
317 const std::array<size_t, N> &end_idx, const size_t &direction,
318 std::array<size_t, N> &strides, const size_t &math_stride,
319 const std::string &interp_func, QuantizeFunc &&quantize_func) {
320 for (size_t i = 0; i < N; i++) {
321 if (end_idx[i] < begin_idx[i]) return 0;
322 }
323 size_t math_begin_idx = begin_idx[direction], math_end_idx = end_idx[direction];
324 size_t n = (math_end_idx - math_begin_idx) / math_stride + 1;
325 if (n <= 1) {
326 return 0;
327 }
328 double predict_error = 0.0;
329 size_t offset = 0;
330 size_t stride = math_stride * original_dim_offsets[direction];
331 std::array<size_t, N> begins, ends, dim_offsets;
332 for (size_t i = 0; i < N; i++) {
333 begins[i] = 0;
334 ends[i] = end_idx[i] - begin_idx[i] + 1;
335 dim_offsets[i] = original_dim_offsets[i];
336 offset += original_dim_offsets[i] * begin_idx[i];
337 }
338 dim_offsets[direction] = stride;
339 size_t stride2x = 2 * stride;
340 if (interp_func == "linear") {
341 begins[direction] = 1;
342 ends[direction] = n - 1;
343 strides[direction] = 2;
344 foreach
345 <T, N>(data, offset, begins, ends, strides, dim_offsets,
346 [&](T *d) { quantize_func(d - data, *d, interp_linear(*(d - stride), *(d + stride))); });
347 if (n % 2 == 0) {
348 begins[direction] = n - 1;
349 ends[direction] = n;
350 foreach
351 <T, N>(data, offset, begins, ends, strides, dim_offsets, [&](T *d) {
352 if (n < 3)
353 quantize_func(d - data, *d, *(d - stride));
354 else
355 quantize_func(d - data, *d, interp_linear1(*(d - stride2x), *(d - stride)));
356 });
357 }
358 } else {
359 size_t stride3x = 3 * stride;
360 size_t i_start = 3;
361 begins[direction] = i_start;
362 ends[direction] = (n >= 3) ? (n - 3) : 0;
363 strides[direction] = 2;
364 foreach
365 <T, N>(data, offset, begins, ends, strides, dim_offsets, [&](T *d) {
366 quantize_func(d - data, *d,
367 interp_cubic(*(d - stride3x), *(d - stride), *(d + stride), *(d + stride3x)));
368 });
369 std::vector<size_t> boundaries;
370 boundaries.push_back(1);
371 if (n % 2 == 1 && n > 3) {
372 boundaries.push_back(n - 2);
373 }
374 if (n % 2 == 0 && n > 4) {
375 boundaries.push_back(n - 3);
376 }
377 if (n % 2 == 0 && n > 2) {
378 boundaries.push_back(n - 1);
379 }
380 for (auto boundary : boundaries) {
381 begins[direction] = boundary;
382 ends[direction] = boundary + 1;
383 foreach
384 <T, N>(data, offset, begins, ends, strides, dim_offsets, [&](T *d) {
385 if (boundary >= 3) {
386 if (boundary + 3 < n)
387 quantize_func(
388 d - data, *d,
389 interp_cubic(*(d - stride3x), *(d - stride), *(d + stride), *(d + stride3x)));
390 else if (boundary + 1 < n)
391 quantize_func(d - data, *d,
392 interp_quad_2(*(d - stride3x), *(d - stride), *(d + stride)));
393 else
394 quantize_func(d - data, *d, interp_linear1(*(d - stride3x), *(d - stride)));
395 } else {
396 if (boundary + 3 < n)
397 quantize_func(d - data, *d,
398 interp_quad_1(*(d - stride), *(d + stride), *(d + stride3x)));
399 else if (boundary + 1 < n)
400 quantize_func(d - data, *d, interp_linear(*(d - stride), *(d + stride)));
401 else
402 quantize_func(d - data, *d, *(d - stride));
403 }
404 });
405 }
406 }
407 return predict_error;
408 }

410 template <class QuantizeFunc>
411 double interpolation(T *data, std::array<size_t, N> begin, std::array<size_t, N> end,
412 const std::string &interp_func, QuantizeFunc &&quantize_func, const int direction,
413 size_t stride = 1) {
414 if constexpr (N == 1) { // old API
415 return interpolation_1d(data, begin[0], end[0], stride, interp_func, quantize_func);
416 } else if constexpr (N == 2) { // old API
417 double predict_error = 0;
418 size_t stride2x = stride * 2;
419 const std::array<int, N> dims = dim_sequences[direction];
420 for (size_t j = (begin[dims[1]] ? begin[dims[1]] + stride2x : 0); j <= end[dims[1]]; j += stride2x) {
421 size_t begin_offset =
422 begin[dims[0]] * original_dim_offsets[dims[0]] + j * original_dim_offsets[dims[1]];
423 predict_error += interpolation_1d(
424 data, begin_offset, begin_offset + (end[dims[0]] - begin[dims[0]]) * original_dim_offsets[dims[0]],
425 stride * original_dim_offsets[dims[0]], interp_func, quantize_func);
426 }
427 for (size_t i = (begin[dims[0]] ? begin[dims[0]] + stride : 0); i <= end[dims[0]]; i += stride) {
428 size_t begin_offset =
429 i * original_dim_offsets[dims[0]] + begin[dims[1]] * original_dim_offsets[dims[1]];
430 predict_error += interpolation_1d(
431 data, begin_offset, begin_offset + (end[dims[1]] - begin[dims[1]]) * original_dim_offsets[dims[1]],
432 stride * original_dim_offsets[dims[1]], interp_func, quantize_func);
433 }
434 return predict_error;
435 } else if constexpr (N == 3 || N == 4) { // new API (for faster speed)
436 double predict_error = 0;
437 size_t stride2x = stride * 2;
438 const std::array<int, N> dims = dim_sequences[direction];
439 std::array<size_t, N> strides;
440 std::array<size_t, N> begin_idx = begin, end_idx = end;
441 strides[dims[0]] = 1;
442 for (uint i = 1; i < N; i++) {
443 begin_idx[dims[i]] = (begin[dims[i]] ? begin[dims[i]] + stride2x : 0);
444 strides[dims[i]] = stride2x;
445 }
446
447 predict_error += interpolation_1d_fastest_dim_first(data, begin_idx, end_idx, dims[0], strides, stride,
448 interp_func, quantize_func);
449 for (uint i = 1; i < N; i++) {
450 begin_idx[dims[i]] = begin[dims[i]];
451 begin_idx[dims[i - 1]] = (begin[dims[i - 1]] ? begin[dims[i - 1]] + stride : 0);
452 strides[dims[i - 1]] = stride;
453 predict_error += interpolation_1d_fastest_dim_first(data, begin_idx, end_idx, dims[i], strides, stride,
454 interp_func, quantize_func);
455 }
456 return predict_error;
457 } else {
458 throw std::runtime_error("Unsupported dimension in InterpolationDecomposition");
459 }
460 }
