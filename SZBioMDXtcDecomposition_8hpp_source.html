91 /* Start from the last frame and look for frames filled with the same value.
92 * Those frames do not need to be compressed - they can all just be filled. */
93 std::tuple<size_t, T> findFillValueAndFirstFilledFrame(T* data, std::vector<size_t> dims) {
94 size_t numDims = dims.size();
95 if (numDims < 3) {
96 return std::make_tuple(dims[0], 0);
97 }
98 size_t frameStride = dims[1] * dims[2];
99 size_t firstFillFrame = dims[0];
100
101 if (firstFillFrame == 0) {
102 return std::make_tuple(firstFillFrame, 0);
103 }
104 /* Assume that the first value of the last frame is the potential fill value */
105 T fillFrameValue = data[(dims[0] - 1) * frameStride];
106 /* To simplify compression/decompression below, assume that the first frame needs
107 * to be compressed. */
108 for (size_t i = dims[0] - 1; i > 0; i--) {
109 size_t idx = i * frameStride;
110 bool allFrameValuesAreFillValue = true;
111
112 for (size_t j = 0; j < dims[1] * dims[2]; j++) {
113 size_t idy = idx + j;
114 if (data[idy] != fillFrameValue) {
115 allFrameValuesAreFillValue = false;
116 break;
117 }
118 }
119 if (allFrameValuesAreFillValue) {
120 firstFillFrame = i;
121 } else {
122 break;
123 }
124 }
125 return std::make_tuple(firstFillFrame, fillFrameValue);
126 }

128 /* This just converts float to integer based on the absolute error. */
129 std::vector<int> compressMultiFrame(T* data) {
130 auto dims = conf.dims;
131 std::vector<size_t> stride({dims[1] * dims[2], dims[2], 1});
132
133 /* Find out if the last frames are all filled with the same value. */
134 std::tuple<size_t, T> fillValueSettings = findFillValueAndFirstFilledFrame(data, dims);
135 firstFillFrame_ = std::get<0>(fillValueSettings);
136 fillValue_ = std::get<1>(fillValueSettings);
137 size_t lastFrame = std::min(dims[0], firstFillFrame_);
138 std::vector<int> quantData(lastFrame * dims[1] * dims[2]);
139
140 for (size_t i = 0; i < lastFrame; i++) {
141 // time
142 for (size_t j = 0; j < dims[1]; j++) {
143 // atoms
144 for (size_t k = 0; k < dims[2]; k++) {
145 // xyz
146 size_t idx = i * stride[0] + j * stride[1] + k;
147 quantData[idx] = quantizer.quantize_and_overwrite(data[idx], 0) - XTC_radius;
148 }
149 }
150 }
151 quantizer.postcompress_data();
152
153 return quantData;
154 }

156 /* This just converts integer to float based on the absolute error. */
157 T* decompressMultiFrame(std::vector<int>& quantData, T* decData) {
158 // printf("Decompressing 3D.\n");
159 auto dims = conf.dims;
160 std::vector<size_t> stride({dims[1] * dims[2], dims[2], 1});
161
162 size_t lastFrame = std::min(dims[0], firstFillFrame_);
163
164 for (size_t i = 0; i < lastFrame; i++) {
165 // time
166 for (size_t j = 0; j < dims[1]; j++) {
167 // atoms
168 for (size_t k = 0; k < dims[2]; k++) {
169 // xyz
170 size_t idx = i * stride[0] + j * stride[1] + k;
171 decData[idx] = quantizer.recover(0, quantData[idx] + XTC_radius);
172 }
173 }
174 }
175 quantizer.postdecompress_data();
176
177 /* Fill frames at the end with the fill value. */
178 for (size_t i = firstFillFrame_; i < dims[0]; i++) {
179 size_t idx = i * stride[0];
180 for (size_t j = 0; j < dims[1] * dims[2]; j++) {
181 size_t idy = idx + j;
182 decData[idy] = fillValue_;
183 }
184 }
185 return decData;
186 }
