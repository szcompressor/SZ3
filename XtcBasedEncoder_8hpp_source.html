29// Integers above 2^24 do not have unique representations in
30// 32-bit floats ie with 24 bits of precision. We use maxAbsoluteInt
31// to check that float values can be transformed into an in-range
32// 32-bit integer. There is no need to ensure we are within the range
33// of ints with exact floating-point representations. However, we should
34// reject all floats above that which converts to an in-range 32-bit integer.
35const float maxAbsoluteInt = std::nextafterf(float(INT_MAX), 0.F); // NOLINT(cert-err58-cpp)

72static void sendbits(struct DataBuffer *buffer, int num_of_bits, int num) {
73 unsigned int lastbyte;
74 int lastbits;
75
76 lastbits = buffer->lastbits;
77 lastbyte = buffer->lastbyte;
78 while (num_of_bits >= CHAR_BIT) {
79 lastbyte = (lastbyte << CHAR_BIT) | ((num >> (num_of_bits - CHAR_BIT)) /* & 0xff*/);
80 buffer->data[buffer->index++] = lastbyte >> lastbits;
81 num_of_bits -= CHAR_BIT;
82 }
83 if (num_of_bits > 0) {
84 lastbyte = (lastbyte << num_of_bits) | num;
85 lastbits += num_of_bits;
86 if (lastbits >= CHAR_BIT) {
87 lastbits -= CHAR_BIT;
88 buffer->data[buffer->index++] = lastbyte >> lastbits;
89 }
90 }
91 buffer->lastbits = lastbits;
92 buffer->lastbyte = lastbyte;
93 if (lastbits > 0) {
94 buffer->data[buffer->index] = lastbyte << (CHAR_BIT - lastbits);
95 }
96}

125static int sizeofints(const int num_of_ints, const unsigned int sizes[]) {
126 int i, num;
127 int bytes[32];
128 unsigned int num_of_bytes, num_of_bits, bytecnt, tmp;
129 num_of_bytes = 1;
130 bytes[0] = 1;
131 num_of_bits = 0;
132 for (i = 0; i < num_of_ints; i++) {
133 tmp = 0;
134 for (bytecnt = 0; bytecnt < num_of_bytes; bytecnt++) {
135 tmp = bytes[bytecnt] * sizes[i] + tmp;
136 bytes[bytecnt] = tmp & 0xff;
137 tmp >>= CHAR_BIT;
138 }
139 while (tmp != 0) {
140 bytes[bytecnt++] = tmp & 0xff;
141 tmp >>= CHAR_BIT;
142 }
143 num_of_bytes = bytecnt;
144 }
145 num = 1;
146 num_of_bytes--;
147 while (bytes[num_of_bytes] >= num) {
148 num_of_bits++;
149 num *= 2;
150 }
151 return num_of_bits + num_of_bytes * CHAR_BIT;
152}

167static void sendints(struct DataBuffer *buffer, const int num_of_ints, const int num_of_bits, unsigned int sizes[],
168 unsigned int nums[]) {
169 int i, num_of_bytes, bytecnt;
170 unsigned int bytes[32], tmp;
171
172 tmp = nums[0];
173 num_of_bytes = 0;
174 do {
175 bytes[num_of_bytes++] = tmp & 0xff;
176 tmp >>= CHAR_BIT;
177 } while (tmp != 0);
178
179 for (i = 1; i < num_of_ints; i++) {
180 if (nums[i] >= sizes[i]) {
181 fprintf(stderr,
182 "major breakdown in sendints num %u doesn't "
183 "match size %u\n",
184 nums[i], sizes[i]);
185 exit(1);
186 }
187 /* use one step multiply */
188 tmp = nums[i];
189 for (bytecnt = 0; bytecnt < num_of_bytes; bytecnt++) {
190 tmp = bytes[bytecnt] * sizes[i] + tmp;
191 bytes[bytecnt] = tmp & 0xff;
192 tmp >>= CHAR_BIT;
193 }
194 while (tmp != 0) {
195 bytes[bytecnt++] = tmp & 0xff;
196 tmp >>= CHAR_BIT;
197 }
198 num_of_bytes = bytecnt;
199 }
200 if (num_of_bits >= num_of_bytes * CHAR_BIT) {
201 for (i = 0; i < num_of_bytes; i++) {
202 sendbits(buffer, CHAR_BIT, bytes[i]);
203 }
204 sendbits(buffer, num_of_bits - num_of_bytes * CHAR_BIT, 0);
205 } else {
206 for (i = 0; i < num_of_bytes - 1; i++) {
207 sendbits(buffer, CHAR_BIT, bytes[i]);
208 }
209 sendbits(buffer, num_of_bits - (num_of_bytes - 1) * CHAR_BIT, bytes[i]);
210 }

220static int receivebits(struct DataBuffer *buffer, int num_of_bits) {
221 int num, lastbits;
222 unsigned int lastbyte;
223 int mask = (1 << num_of_bits) - 1;
224
225 lastbits = buffer->lastbits;
226 lastbyte = buffer->lastbyte;
227
228 num = 0;
229 while (num_of_bits >= CHAR_BIT) {
230 lastbyte = (lastbyte << CHAR_BIT) | buffer->data[buffer->index++];
231 num |= (lastbyte >> lastbits) << (num_of_bits - CHAR_BIT);
232 num_of_bits -= CHAR_BIT;
233 }
234 if (num_of_bits > 0) {
235 if (lastbits < num_of_bits) {
236 lastbits += CHAR_BIT;
237 lastbyte = (lastbyte << CHAR_BIT) | buffer->data[buffer->index++];
238 }
239 lastbits -= num_of_bits;
240 num |= (lastbyte >> lastbits) & ((1 << num_of_bits) - 1);
241 }
242 num &= mask;
243 buffer->lastbits = lastbits;
244 buffer->lastbyte = lastbyte;
245 return num;
246}

257static void receiveints(struct DataBuffer *buffer, const int num_of_ints, int num_of_bits, const unsigned int sizes[],
258 int nums[]) {
259 int bytes[32];
260 int i, j, num_of_bytes, p, num;
261
262 bytes[0] = bytes[1] = bytes[2] = bytes[3] = 0;
263 num_of_bytes = 0;
264 while (num_of_bits > CHAR_BIT) {
265 bytes[num_of_bytes++] = receivebits(buffer, CHAR_BIT);
266 num_of_bits -= CHAR_BIT;
267 }
268 if (num_of_bits > 0) {
269 bytes[num_of_bytes++] = receivebits(buffer, num_of_bits);
270 }
271 for (i = num_of_ints - 1; i > 0; i--) {
272 num = 0;
273 for (j = num_of_bytes - 1; j >= 0; j--) {
274 num = (num << CHAR_BIT) | bytes[j];
275 p = num / sizes[i];
276 bytes[j] = p;
277 num = num - p * sizes[i];
278 }
279 nums[i] = num;
280 }
281 nums[0] = bytes[0] | (bytes[1] << CHAR_BIT) | (bytes[2] << 16) | (bytes[3] << 24);
282}
