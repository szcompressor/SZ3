21unsigned int static round_up_power_of_2(unsigned int base) {
22 base -= 1;
23 base = base | (base >> 1);
24 base = base | (base >> 2);
25 base = base | (base >> 4);
26 base = base | (base >> 8);
27 base = base | (base >> 16);
28 return base + 1;
29}

31unsigned int static inline estimate_quantization_intervals(const std::vector<size_t> &intervals, size_t sample_count) {
32 size_t target = sample_count * QuantIntvAccThreshold;
33 size_t sum = 0;
34 size_t i = 0;
35 for (i = 0; i < intervals.size(); i++) {
36 sum += intervals[i];
37 if (sum > target) break;
38 }
39 if (i == intervals.size()) i = intervals.size() - 1;
40 unsigned int accIntervals = 2 * (i + 1);
41 unsigned int num_intervals = 2 * round_up_power_of_2(accIntervals);
42 return (num_intervals > 32) ? num_intervals : 32;
43}

45template <typename T>
46float estimate_mean_freq_and_position(const std::vector<size_t> &freq_intervals, double precision, size_t sample_count,
47 T &mean_guess) {
48 size_t max_sum = 0;
49 size_t max_index = 0;
50 size_t tmp_sum = 0;
51 for (size_t i = 1; i < freq_intervals.size() - 2; i++) {
52 tmp_sum = freq_intervals[i] + freq_intervals[i + 1];
53 if (tmp_sum > max_sum) {
54 max_sum = tmp_sum;
55 max_index = i;
56 }
57 }
58 mean_guess += precision * static_cast<ptrdiff_t>(max_index + 1 - (freq_intervals.size() >> 1));
59 return max_sum * 1.0 / sample_count;
60}

63float sample_rough_mean_3d(const T *data, size_t r1, size_t r2, size_t r3, size_t sample_distance) {
64 double mean = 0;
65 size_t len = r1 * r2 * r3;
66 const T *data_pos = data;
67 size_t offset_count = 0;
68 size_t offset_count_2 = 0;
69 size_t mean_count = 0;
70 while (data_pos - data < len) {
71 mean += *data_pos;
72 mean_count++;
73 data_pos += sample_distance;
74 offset_count += sample_distance;
75 offset_count_2 += sample_distance;
76 if (offset_count >= r3) {
77 offset_count = 0;

96int optimize_quant_invl_3d(const T *data, size_t r1, size_t r2, size_t r3, double precision, float &pred_freq,

102 size_t sample_count = 0;
103 size_t sample_distance = QuantIntvSampleDistance;
104 size_t offset_count = sample_distance - 2; // count r3 offset
105 size_t offset_count_2 = 0;
106 size_t r23 = r2 * r3;
107 size_t len = r1 * r23;
108 const T *data_pos = data + r23 + r3 + offset_count;
109 size_t n1_count = 1, n2_count = 1; // count i,j sum
110 T pred_value = 0;
111 double mean_diff = 0;
112 ptrdiff_t freq_index = 0;
113 size_t pred_index = 0;
114 float pred_err = 0;

116 while (data_pos - data < len) {
117 pred_value = lorenzo_predict_3d(data_pos, r23, r3);
118 pred_err = fabs(pred_value - *data_pos);
119 if (pred_err < precision) freq_count++;
120 pred_index = (pred_err / precision + 1) / 2;
121 if (pred_index >= intervals.size()) {
122 pred_index = intervals.size() - 1;
123 }
124 intervals[pred_index]++;
125
126 mean_diff = *data_pos - mean_rough;
127 if (mean_diff > 0)
128 freq_index = static_cast<ptrdiff_t>(mean_diff / precision) + radius;
129 else
130 freq_index = static_cast<ptrdiff_t>(mean_diff / precision) - 1 + radius;
131 if (freq_index <= 0) {
132 freq_intervals[0]++;
133 } else if (freq_index >= static_cast<ptrdiff_t>(freq_intervals.size())) {
134 freq_intervals[freq_intervals.size() - 1]++;
135 } else {
136 freq_intervals[freq_index]++;
137 }
138 offset_count += sample_distance;
139 if (offset_count >= r3) {
140 n2_count++;
141 if (n2_count == r2) {
142 n1_count++;
143 n2_count = 1;
144 data_pos += r3;
145 }
146 offset_count_2 = (n1_count + n2_count) % sample_distance;
147 data_pos += (r3 + sample_distance - offset_count) + (sample_distance - offset_count_2);
148 offset_count = (sample_distance - offset_count_2);
149 if (offset_count == 0) offset_count++;
150 } else
151 data_pos += sample_distance;
152 sample_count++;
153 }
154 pred_freq = freq_count * 1.0 / sample_count;
155 mean_guess = mean_rough;
156 mean_freq = estimate_mean_freq_and_position(freq_intervals, precision, sample_count, mean_guess);
157 return estimate_quantization_intervals(intervals, sample_count);
158}
