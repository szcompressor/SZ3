28 std::vector<int> compress(const Config& conf, T* data) override {
29 // 1. SVD Decomposition (ST-HOSVD)
30 Eigen::array<Eigen::Index, N> tensor_dims;

32 tensor_dims[i] = conf.dims[i];
33 }

37 std::vector<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>> temp_factor_matrices;

39 T tau = config.svd_energy_threshold;

45 auto U = rsvd_res.first;

47 temp_factor_matrices.push_back(U);

63 factor_dims.resize(temp_factor_matrices.size());

166 // From paper Algorithm 2: Adaptive rSVD (with Rank Finding)
167 std::pair<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>, Eigen::VectorX<T>>
168 adaptive_rsvd(const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>& A, T tau) {
169 T energy_A = A.squaredNorm();
170 if (energy_A == 0) {
171 return {Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Zero(A.rows(), 1), Eigen::VectorX<T>::Zero(1)};
172 }
173 long min_dim = std::min(A.rows(), A.cols());
174
175 // Predefined sequence of ranks to test
176 std::vector<int> ranks_to_test;
177 for (int r = 8; r < min_dim; r *= 2) {
178 ranks_to_test.push_back(r);
179 }
180 ranks_to_test.push_back(min_dim);
181
182 for (int rank : ranks_to_test) {
183 auto rsvd_res = perform_rsvd(A, rank);
184 auto sigma = rsvd_res.second;
185 T captured_energy = sigma.squaredNorm();
186 if (captured_energy / energy_A >= tau) {
187 return rsvd_res;
188 }
189 }
190 // Fallback: return result of largest rank tested
191 return perform_rsvd(A, min_dim);
192 }

194 std::pair<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>, Eigen::VectorX<T>>
195 perform_rsvd(const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>& A, int rank) {
196 // unfolded matrix is short and wide, so A.rows() < A.cols()
197 rank = std::min({rank, (int)A.rows(), (int)A.cols()});
198
199 if (rank >= A.rows()) {
200 Eigen::JacobiSVD<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>> svd(A, Eigen::ComputeThinU | Eigen::ComputeThinV);
201 return {svd.matrixU(), svd.singularValues()};
202 }
203
204 int l = rank + config.svd_oversampling_param;
205 l = std::min({l, (int)A.rows(), (int)A.cols()});
206
207 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> Omega = Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Random(A.cols(), l);
208 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> Y = A * Omega;
209 Eigen::HouseholderQR<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>> qr(Y);
210 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> Q = qr.householderQ() * Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>::Identity(Y.rows(), l);
211
212 // Gram-based rSVD from paper's Algorithm 1
213 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> B = (Q.transpose() * A) * (Q.transpose() * A).transpose();
214
215 Eigen::SelfAdjointEigenSolver<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>> eigen_solver(B);
216 auto eigenvalues = eigen_solver.eigenvalues();
217 auto eigenvectors = eigen_solver.eigenvectors();
218
219 // Sort eigenvalues and eigenvectors in descending order
220 std::vector<std::pair<T, Eigen::VectorX<T>>> eigen_pairs;
221 for (int i = 0; i < eigenvalues.size(); ++i) {
222 eigen_pairs.push_back({eigenvalues(i), eigenvectors.col(i)});
223 }
224 std::sort(eigen_pairs.begin(), eigen_pairs.end(), [](const auto& a, const auto& b) {
225 return a.first > b.first;
226 });
227
228 Eigen::Vector<T, Eigen::Dynamic> sorted_eigenvalues(l);
229 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> U0(l, l);
230
231 for (int i = 0; i < l; ++i) {
232 sorted_eigenvalues(i) = eigen_pairs[i].first > 0 ? eigen_pairs[i].first : 0;
233 U0.col(i) = eigen_pairs[i].second;
234 }
235
236 Eigen::Vector<T, Eigen::Dynamic> sigma = sorted_eigenvalues.cwiseSqrt();
237 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> U = Q * U0;
238
239 return {U.leftCols(rank), sigma.head(rank)};
240 }

248 Eigen::Tensor<T, N> contract_and_shuffle(const Eigen::Tensor<T, N>& tensor, const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>& matrix, int mode, bool transpose_matrix) {
249 Eigen::Tensor<T, 2> matrix_tensor = transpose_matrix ? matrix_to_tensor(matrix.transpose()) : matrix_to_tensor(matrix);
250
251 Eigen::array<Eigen::IndexPair<int>, 1> product_dims = { Eigen::IndexPair<int>(mode, 1) };
252
253 auto contracted_tensor = tensor.contract(matrix_tensor, product_dims);
254
255 std::vector<int> p(N);
256 std::iota(p.begin(), p.end(), 0);
257 int last = p.back();
258 p.pop_back();
259 p.insert(p.begin() + mode, last);
260
261 Eigen::array<int, N> shuffle_perm;
262 for(int i=0; i<N; ++i) shuffle_perm[i] = p[i];
263
264 return contracted_tensor.shuffle(shuffle_perm);
265 }

268 Eigen::Tensor<T, N> dequantize_and_reconstruct() {
269 Eigen::array<Eigen::Index, N> eigen_core_dims;
270 for(int i=0; i<N; ++i) eigen_core_dims[i] = core_dims[i];
271 Eigen::Tensor<T, N> core_tensor(eigen_core_dims);
272 for(size_t i=0; i<quantized_core.size(); ++i) {
273 core_tensor.data()[i] = svd_quantizer.recover(0, quantized_core[i]);
274 }
275
276 std::vector<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>> factor_matrices(quantized_factors.size());
277 for(size_t i=0; i<quantized_factors.size(); ++i) {
278 factor_matrices[i].resize(factor_dims[i].first, factor_dims[i].second);
279 for(size_t j=0; j<quantized_factors[i].size(); ++j) {
280 factor_matrices[i].data()[j] = svd_quantizer.recover(0, quantized_factors[i][j]);
281 }
282 }
283
284 Eigen::Tensor<T, N> reconstructed_tensor = core_tensor;
285 for (int mode = N - 1; mode >= 0; --mode) {
286 reconstructed_tensor = contract_and_shuffle(reconstructed_tensor, factor_matrices[mode], mode, false);
287 }
288 return reconstructed_tensor;
289 }

291 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> unfold(const Eigen::Tensor<T, N>& tensor, int mode) {
292 Eigen::array<int, N> shuffle;
293 std::iota(shuffle.begin(), shuffle.end(), 0);
294 std::swap(shuffle[0], shuffle[mode]);
295 auto shuffled_tensor = tensor.shuffle(shuffle);
296
297 long rows = tensor.dimension(mode);
298 long cols = tensor.size() / rows;
299
300 Eigen::array<Eigen::Index, 2> unfolded_dims = {rows, cols};
301 Eigen::Tensor<T, 2> unfolded_tensor = shuffled_tensor.reshape(unfolded_dims);
302
303 return Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>(Eigen::Map<Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>>(unfolded_tensor.data(), rows, cols));
304 }
