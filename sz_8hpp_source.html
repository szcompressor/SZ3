43template <class T>
44size_t SZ_compress(const SZ3::Config& config, const T* data, char* cmpData, size_t cmpCap) {
45 using namespace SZ3;
46 Config conf(config);
47
48 if (cmpCap < SZ_compress_size_bound<T>(conf)) {
49 throw std::invalid_argument(SZ3_ERROR_COMP_BUFFER_NOT_LARGE_ENOUGH);
50 }
51
52 auto cmpDataPos = reinterpret_cast<uchar*>(cmpData);
53
54 // save 16 bytes header
55 write(config.sz3MagicNumber, cmpDataPos); // magic number (4 bytes)
56 write(config.sz3DataVer, cmpDataPos); // data version (4 bytes)
57 auto sizeHeaderPos = cmpDataPos;
58 cmpDataPos += 8; // reserve space for cmp data size (8 bytes)
59
60 // begin compression
61 auto cmpDataCap = cmpCap - 16 - conf.size_est() * 2;
62 uint64_t cmpDataSize = 0;
63 if (conf.N == 1) {
64 cmpDataSize = SZ_compress_impl<T, 1>(conf, data, cmpDataPos, cmpDataCap);
65 } else if (conf.N == 2) {
66 cmpDataSize = SZ_compress_impl<T, 2>(conf, data, cmpDataPos, cmpDataCap);
67 } else if (conf.N == 3) {
68 cmpDataSize = SZ_compress_impl<T, 3>(conf, data, cmpDataPos, cmpDataCap);
69 } else if (conf.N == 4) {
70 cmpDataSize = SZ_compress_impl<T, 4>(conf, data, cmpDataPos, cmpDataCap);
71 } else {
72 throw std::invalid_argument("Data dimension higher than 4 is not supported.");
73 }
74
75 // save compressed data size back in header
76 write(cmpDataSize, sizeHeaderPos);
77
78 // save config
79 cmpDataPos += cmpDataSize;
80 auto confSize = conf.save(cmpDataPos);
81
82 return 16 + cmpDataSize + confSize;
83}
