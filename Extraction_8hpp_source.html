14template <uint N>
15float cal_sampling_ratio(size_t block, size_t n, size_t dmin, std::vector<size_t> dims) {
16 size_t sample_n = 1;
17 for (auto dim : dims) {
18 sample_n *= dim / dmin * 2 * block;
19 }
20 return sample_n * 1.0 / n;
21}

23template <class T, uint N>
24inline typename std::enable_if<N == 4, std::vector<T>>::type sampling(T *data, std::vector<size_t> dims,
25 size_t &sample_num,
26 std::vector<size_t> &sample_dims,
27 size_t &sampling_block) {
28 assert(dims.size() == N);
29 assert(sample_dims.size() == N);
30 Timer timer(true);
31 size_t num = std::accumulate(dims.begin(), dims.end(), static_cast<size_t>(1), std::multiplies<size_t>());
32
33 size_t dmin = *std::min_element(dims.begin(), dims.end());
34 sampling_block = dmin;
35 while (cal_sampling_ratio<N>(sampling_block, num, dmin, dims) > 0.035) {
36 sampling_block--;
37 }
38 if (sampling_block * 2 > dmin) {
39 sampling_block = dmin / 2;
40 }
41 if (sampling_block < 9) {
42 sample_dims = dims;
43 sample_num = num;
44 return std::vector<T>();
45 }
46 size_t b0 = dims[0] / dmin;
47 size_t b1 = dims[1] / dmin;
48 size_t b2 = dims[2] / dmin;
49 size_t b3 = dims[3] / dmin;
50 sample_dims[0] = b0 * 2 * sampling_block;
51 sample_dims[1] = b1 * 2 * sampling_block;
52 sample_dims[2] = b2 * 2 * sampling_block;
53 sample_dims[3] = b3 * 2 * sampling_block;
54 size_t di, dj, dk, dt;
55 sample_num = sample_dims[0] * sample_dims[1] * sample_dims[2] * sample_dims[3];
56 std::vector<T> sampling_data(sample_num, 0);
57
58 for (size_t bi = 0; bi < b0; bi++) {
59 for (size_t bj = 0; bj < b1; bj++) {
60 for (size_t bk = 0; bk < b2; bk++) {
61 for (size_t bt = 0; bt < b3; bt++) {
62 for (size_t i = 0; i < 2 * sampling_block; i++) {
63 for (size_t j = 0; j < 2 * sampling_block; j++) {
64 for (size_t k = 0; k < 2 * sampling_block; k++) {
65 for (size_t t = 0; t < 2 * sampling_block; t++) {
66 di = i < sampling_block ? i + sampling_block : dmin - 3 * sampling_block + i;
67 dj = j < sampling_block ? j + sampling_block : dmin - 3 * sampling_block + j;
68 dk = k < sampling_block ? k + sampling_block : dmin - 3 * sampling_block + k;
69 dt = t < sampling_block ? t + sampling_block : dmin - 3 * sampling_block + t;
70 size_t idx = (bi * dmin + di) * dims[1] * dims[2] * dims[3] +
71 (bj * dmin + dj) * dims[2] * dims[3] + (bk * dmin + dk) * dims[3] +
72 bt * dmin + dt;
73 sampling_data[(bi * 2 * sampling_block + i) * sample_dims[1] * sample_dims[2] *
74 sample_dims[3] +
75 (bj * 2 * sampling_block + j) * sample_dims[1] * sample_dims[2] +
76 (bk * 2 * sampling_block + k) * sample_dims[2] +
77 bt * 2 * sampling_block + t] = data[idx];
78 }
79 }
80 }
81 }
82 }
83 }
84 }
85 }
