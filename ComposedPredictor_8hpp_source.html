31 bool precompress(const block_iter &block) override {
32 std::vector<double> predict_error(predictors.size(), 0);
33 std::vector<bool> isvalid(predictors.size());
34 for (size_t i = 0; i < predictors.size(); i++) {
35 isvalid[i] = predictors[i]->precompress(block);
36 if (isvalid[i]) {
37 block_iter::foreach_sampling(block, [&](T *c, const std::array<size_t, N> &index) {
38 predict_error[i] += predictors[i]->estimate_error(block, c, index);
39 });
40 } else {
41 predict_error[i] = std::numeric_limits<double>::max();
42 }
43 }
44 sid = std::distance(predict_error.begin(), std::min_element(predict_error.begin(), predict_error.end()));
45 return isvalid[sid];
46 }
