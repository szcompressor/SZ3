28 RegressionPredictor(uint block_size, double eb)

34 bool precompress(const block_iter &block) override {
35 auto range = block.get_block_range();
36
37 std::array<double, N> dims{0};
38 double num_elements = 1;
39 for (uint i = 0; i < N; i++) {
40 dims[i] = range[i].second - range[i].first;
41 if (dims[i] <= 1) {
42 return false;
43 }
44 num_elements *= dims[i];
45 }
46
47 std::array<double, N + 1> sum{0};
48 block_iter::foreach (block, [&](T *c, const std::array<size_t, N> &index) {
49 for (uint i = 0; i < N; i++) {
50 sum[i] += index[i] * (*c);
51 }
52 sum[N] += *c;
53 });
54 std::fill(current_coeffs.begin(), current_coeffs.end(), 0);
55 current_coeffs[N] = sum[N] / num_elements;
56 for (uint i = 0; i < N; i++) {
57 current_coeffs[i] = (2 * sum[i] / (dims[i] - 1) - sum[N]) * 6 / num_elements / (dims[i] + 1);
58 current_coeffs[N] -= (dims[i] - 1) * current_coeffs[i] / 2;
59 }
60 return true;
61 }
62
63 void precompress_block_commit() noexcept override {
64 pred_and_quantize_coefficients();

83 ALWAYS_INLINE T predict(const block_iter &block, T *d, const std::array<size_t, N> &index) override {
84 if constexpr (N == 1) {
85 return current_coeffs[0] * index[0] + current_coeffs[1];
86 } else if constexpr (N == 2) {
87 return current_coeffs[0] * index[0] + current_coeffs[1] * index[1] + current_coeffs[2];
88 } else if constexpr (N == 3) {
89 return current_coeffs[0] * index[0] + current_coeffs[1] * index[1] + current_coeffs[2] * index[2] +
90 current_coeffs[3];
91 } else if constexpr (N == 4) {
92 return current_coeffs[0] * index[0] + current_coeffs[1] * index[1] + current_coeffs[2] * index[2] +
93 current_coeffs[3] * index[3] + current_coeffs[4];
94 } else {
95 static_assert(N <= 4, "Unsupported dimension or layer configuration");
96 return T(0);

154 void pred_and_quantize_coefficients() {
155 for (int i = 0; i < static_cast<int>(N); i++) {
156 regression_coeff_quant_inds.push_back(

164 for (int i = 0; i < static_cast<int>(N); i++) {
165 current_coeffs[i] =

167 }
168 current_coeffs[N] =
169 quantizer_independent.recover(current_coeffs[N], regression_coeff_quant_inds[regression_coeff_index++]);
170 }
