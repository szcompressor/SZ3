64template <typename T, uint N>
65std::shared_ptr<concepts::CompressorInterface<T>> make_sz_timebased(const Config &conf, T *data_ts0) {
66 std::vector<std::shared_ptr<concepts::PredictorInterface<T, N - 1>>> predictors;
67
68 int use_single_predictor = (conf.lorenzo + conf.lorenzo2 + conf.regression) == 1;
69 if (conf.lorenzo) {
70 if (use_single_predictor) {
71 return make_sz_timebased2<T, N>(conf, LorenzoPredictor<T, N - 1, 1>(conf.absErrorBound), data_ts0);
72 } else {
73 predictors.push_back(std::make_shared<LorenzoPredictor<T, N - 1, 1>>(conf.absErrorBound));
74 }
75 }
76 if (conf.lorenzo2) {
77 if (use_single_predictor) {
78 return make_sz_timebased2<T, N>(conf, LorenzoPredictor<T, N - 1, 2>(conf.absErrorBound), data_ts0);
79 } else {
80 predictors.push_back(std::make_shared<LorenzoPredictor<T, N - 1, 2>>(conf.absErrorBound));
81 }
82 }
83 if (conf.regression) {
84 if (use_single_predictor) {
85 return make_sz_timebased2<T, N>(conf, RegressionPredictor<T, N - 1>(conf.blockSize, conf.absErrorBound),
86 data_ts0);
87 } else {
88 predictors.push_back(std::make_shared<RegressionPredictor<T, N - 1>>(conf.blockSize, conf.absErrorBound));
89 }
90 }
91
92 return make_sz_timebased2<T, N>(conf, ComposedPredictor<T, N - 1>(predictors), data_ts0);
93}

125template <typename T, uint N>
126float *VQ(Config conf, size_t ts, T *data, size_t &compressed_size, bool decom, int method, float level_start,
127 float level_offset, int level_num) {
128 if (level_num == 0) {
129 throw std::runtime_error("VQ/VQT not availble on current dataset, please use ADP or MT");
130 }
131
132 auto sz = make_compressor_exaalt<T, N>(LinearQuantizer<float>(conf.absErrorBound, conf.quantbinCnt / 2),
133 HuffmanEncoder<int>(), Lossless_zstd(), method);
134 sz->set_level(level_start, level_offset, level_num);
135
136 Timer timer(true);
137 compressed_size = 2 * conf.num * sizeof(T);
138 auto compressed = static_cast<uchar *>(malloc(compressed_size));
139 sz->compress(conf, data, compressed, compressed_size);
140 total_compress_time += timer.stop("Compression");
141 if (!decom) {
142 free(compressed);
143 return nullptr;
144 }
145 // auto ratio = conf.num * sizeof(T) * 1.0 / compressed_size;
146 // std::cout << "Compression Ratio = " << ratio << std::endl;
147 // std::cout << "Compressed size = " << compressed_size << std::endl;
148
149 timer.start();
150 auto ts_dec_data = new T[conf.num];
151 sz->decompress(conf, compressed, compressed_size, ts_dec_data);
152 total_decompress_time += timer.stop("Decompression");
153
154 free(compressed);
155 return ts_dec_data;
156}

361template <typename T, uint N>
362inline typename std::enable_if<N == 1 || N == 2, size_t>::type MDZ_Compress(Config conf, T *input_data, T *dec_data,
363 size_t batch_size, int method = -1) {
364 // if (N != 2) {
365 // throw std::invalid_argument("dimension should be 2");
366 // }
367 if (batch_size == 0) {
368 batch_size = conf.dims[0];
369 }
370 int method_batch = 0;
371 if (method == -1) {
372 method_batch = 50;
373 }
374 // std::cout << "****************** Options ********************" << std::endl;
375 // std::cout << "dimension = " << N
376 // << ", error bound = " << conf.absErrorBound
377 // << ", method = " << method
378 // << ", method_update_batch = " << method_batch
379 // << ", batch_size = " << batch_size
380 // << ", quan_state_num = " << conf.quantbinCnt
381 // // << ", encoder = " << conf.encoder_op
382 // // << ", lossless = " << conf.lossless_op
383 // << std::endl;
384
385 // auto data_all = readfile<T>(input_path.data(), 0, conf.num);
386 // auto data_all = input_data;
387 auto data_ts0 = std::vector<T>(input_data, input_data + conf.dims[1]);
388
389 float level_start, level_offset;
390 int level_num = 0;
391 if (method != 2 && method != 3 && method != 4) {
392 size_t sample_num = 0.1 * conf.dims[1];
393 sample_num = std::min(sample_num, static_cast<size_t>(20000));
394 sample_num = std::max(sample_num, std::min(static_cast<size_t>(5000), conf.dims[1]));
395 get_cluster(input_data, conf.dims[1], level_start, level_offset, level_num, sample_num);
396 if (level_num > conf.dims[1] * 0.25) {
397 level_num = 0;
398 }
399 // if (level_num != 0) {
400 // printf("start = %.3f , level_offset = %.3f, nlevel=%d\n", level_start, level_offset, level_num);
401 // }
402 }
403
404 auto dims = conf.dims;
405 // auto total_num = conf.num;
406 size_t total_compressed_size = 0;
407 int current_method = method;
408 bool lossless_first_frame = false;
409 for (size_t ts = 0; ts < dims[0]; ts += batch_size) {
410 conf.dims[0] = (ts + batch_size > dims[0] ? dims[0] - ts : batch_size);
411 conf.num = conf.dims[0] * conf.dims[1];
412
413 T *data = &input_data[ts * conf.dims[1]];
414
415 T max = *std::max_element(data, data + conf.num);
416 T min = *std::min_element(data, data + conf.num);
417 if (conf.errorBoundMode == EB_ABS) {
418 conf.relErrorBound = conf.absErrorBound / (max - min);
419 } else if (conf.errorBoundMode == EB_REL) {
420 conf.absErrorBound = conf.relErrorBound * (max - min);
421 }
422
423 // std::cout << "****************** Compression From " << ts << " to " << ts + conf.dims[0] - 1
424 // << " ******************" << std::endl;
425 // std::cout<<method_batch<<" "<<ts<<" "<<conf.batch_size<<" "<<method_batch<<std::endl;
426 if (method_batch > 0 && ts / batch_size % method_batch == 0) {
427 select<T, N>(conf, current_method, ts, input_data, level_start, level_offset, level_num, data_ts0.data(),
428 batch_size);
429 }
430 // printf("Compressor = %s\n", compressor_names[current_method]);
431
432 std::cout << "From " << ts << " to " << ts + conf.dims[0] - 1
433 << " , Compressor = " << compressor_names[current_method] << std::endl;
434
435 T *ts_dec_data;
436 size_t compressed_size;
437 if (current_method == 0) {
438 ts_dec_data =
439 VQ<T, N>(conf, ts, data, compressed_size, true, current_method, level_start, level_offset, level_num);
440 } else if (current_method == 1) {
441 ts_dec_data =
442 VQ<T, N>(conf, ts, data, compressed_size, true, current_method, level_start, level_offset, level_num);
443 } else if (current_method == 2) {
444 lossless_first_frame = true;
445 ts_dec_data = MT<T, N>(conf, ts, data, compressed_size, true, data_ts0.data());
446 } else if (current_method == 4) {
447 ts_dec_data = MT<T, N>(conf, ts, data, compressed_size, true, static_cast<T *>(nullptr));
448 } else {
449 ts_dec_data = SZ2<T, N>(conf, ts, data, compressed_size, true);
450 }
451 total_compressed_size += compressed_size;
452 memcpy(&dec_data[ts * conf.dims[1]], ts_dec_data, conf.num * sizeof(T));
453 }
454 if (lossless_first_frame) {
455 auto zstd = SZ3::Lossless_zstd();
456 size_t inSize = conf.dims[1] * sizeof(T);
457 uchar *buffer = new uchar[inSize * 2];
458 auto cmpSize = zstd.compress(reinterpret_cast<uchar *>(data_ts0.data()), inSize, buffer, inSize * 2);
459 delete[] buffer;
460 // printf("outsize %lu\n", cmpSize);
461 total_compressed_size += cmpSize;
462 }
463
464 return total_compressed_size;
465}

467template <typename T, uint N>
468inline typename std::enable_if<N == 3, size_t>::type MDZ_Compress(Config conf, T *input_data, T *dec_data,
469 size_t batch_size, int method = -1) {
470 size_t total_compressed_size = 0;
471 auto dims = conf.dims;
472 std::vector<T> input(conf.num), output(conf.num);
473 for (size_t frame = 0; frame < conf.dims[0]; frame++) {
474 for (size_t atom = 0; atom < conf.dims[1]; atom++) {
475 for (size_t xyz = 0; xyz < conf.dims[2]; xyz++) {
476 input[xyz * dims[0] * dims[1] + frame * dims[1] + atom] =
477 input_data[frame * dims[1] * dims[2] + atom * dims[2] + xyz];
478 }
479 }
480 }
481 for (size_t i = 0; i < conf.dims[2]; i++) {
482 Config conf_2D(conf);
483 conf_2D.dims = {conf.dims[0], conf.dims[1]};
484 conf_2D.num = conf.dims[0] * conf.dims[1];
485 total_compressed_size +=
486 MDZ_Compress<T, 2>(conf_2D, input.data() + i * conf.dims[0] * conf.dims[1],
487 output.data() + i * conf.dims[0] * conf.dims[1], batch_size, method);
488 }
489 for (size_t frame = 0; frame < conf.dims[0]; frame++) {
490 for (size_t atom = 0; atom < conf.dims[1]; atom++) {
491 for (size_t xyz = 0; xyz < conf.dims[2]; xyz++) {
492 dec_data[frame * dims[1] * dims[2] + atom * dims[2] + xyz] =
493 output[xyz * dims[0] * dims[1] + frame * dims[1] + atom];
494 }
495 }
496 }
497 return total_compressed_size;
498}
